1//! global_digital_address
2//! Square-cell Web Mercator encoder/decoder for globally unique grid codes.
3//!
4//! API:
5//!   - `get_digi_pin(lat, lon, levels)` -> String
6//!   - `get_lat_lng_from_digipin(code)` -> Result<LatLng, Error>
7//!   - constants: `DIGIPIN_GRID`, `MAX_LAT`
8//!   - `approx_cell_size_meters(levels)`
9//!
10//! Notes:
11//!   - Uses Web Mercator (EPSG:3857) math identical to the JS reference.
12//!   - Longitude normalized to (-180, 180]; latitude clamped to ±MAX_LAT.
13//!   - Grid subdivision is 6×6 per level.
14
15use core::f64::consts::PI;
16
17/// WGS-84 Earth radius used by Web Mercator (meters)
18const R: f64 = 6_378_137.0;
19
20/// Mercator latitude clamp (degrees)
21pub const MAX_LAT: f64 = 85.051_128_78;
22
23/// 6×6 symbol grid (rows top→bottom, columns left→right)
24pub const DIGIPIN_GRID: [[char; 6]; 6] = [
25    ['I', 'A', 'B', 'C', 'D', 'E'],
26    ['G', 'H', 'J', 'K', 'L', 'M'],
27    ['N', 'P', 'Q', 'R', 'S', 'T'],
28    ['U', 'r', 'W', 'X', 'Y', 'Z'],
29    ['a', 'b', '9', 'd', 'V', 'F'],
30    ['2', '3', '4', '5', '6', '7'],
31];
32
33#[derive(Debug, Clone, Copy, PartialEq)]
34pub struct LatLng {
35    pub latitude: f64,
36    pub longitude: f64,
37}
38
39#[derive(thiserror::Error, Debug)]
40pub enum Error {
41    #[error("lat/lon must be finite numbers")]
42    NonFiniteInput,
43    #[error("invalid DIGIPIN")]
44    InvalidPin,
45    #[error("invalid character '{0}' in DIGIPIN")]
46    InvalidChar(char),
47}
48
49#[inline]
50fn clamp(v: f64, lo: f64, hi: f64) -> f64 { v.max(lo).min(hi) }
51
52/// Normalize longitude to (-180, 180]; returns -180 instead of +180.
53#[inline]
54fn normalize_lon(lon: f64) -> f64 {
55    let x = ((lon + 180.0) % 360.0 + 360.0) % 360.0 - 180.0;
56    if x == 180.0 { -180.0 } else { x }
57}
58
59#[inline] fn lon_to_x(lon_deg: f64) -> f64 { R * normalize_lon(lon_deg).to_radians() }
60#[inline] fn x_to_lon(x: f64) -> f64 { normalize_lon((x / R).to_degrees()) }
61
62#[inline]
63fn lat_to_y(lat_deg: f64) -> f64 {
64    let phi = clamp(lat_deg, -MAX_LAT, MAX_LAT).to_radians();
65    R * f64::ln(f64::tan(PI / 4.0 + phi / 2.0))
66}
67
68#[inline]
69fn y_to_lat(y: f64) -> f64 {
70    let phi = 2.0 * f64::atan(f64::exp(y / R)) - PI / 2.0;
71    phi.to_degrees()
72}
73
74struct MercBounds { min_x: f64, max_x: f64, min_y: f64, max_y: f64 }
75
76fn merc_bounds() -> MercBounds {
77    let min_x = -PI * R;
78    let max_x =  PI * R;
79    let max_phi = MAX_LAT.to_radians() / 2.0 + PI / 4.0;
80    let max_y = R * f64::ln(f64::tan(max_phi));
81    let min_y = -max_y;
82    MercBounds { min_x, max_x, min_y, max_y }
83}

85/// Encode lat/lon to a square-cell global code.
86/// - `levels`: number of symbols (default = 10)
87pub fn get_digi_pin(lat: f64, lon: f64, levels: usize) -> Result<String, Error> {
88    if !lat.is_finite() || !lon.is_finite() { return Err(Error::NonFiniteInput); }
89
90    let lat = clamp(lat, -MAX_LAT, MAX_LAT);
91    let lon = normalize_lon(lon);
92
93    let mut x = lon_to_x(lon);
94    let mut y = lat_to_y(lat);
95
96    let eps = 1e-9_f64;
97    let MercBounds { mut min_x, mut max_x, mut min_y, mut max_y } = merc_bounds();
98
99    // keep strictly inside bounds
100    x = x.max(min_x + eps).min(max_x - eps);
101    y = y.max(min_y + eps).min(max_y - eps);
102
103    let mut code = String::with_capacity(levels);
104
105    for _ in 0..levels {
106        let x_div = (max_x - min_x) / 6.0;
107        let y_div = (max_y - min_y) / 6.0;
108
109        // top row = 0
110        let row_raw = 5.0 - ((y - min_y) / y_div).floor();
111        let col_raw = ((x - min_x) / x_div).floor();
112
113        let row = clamp(row_raw, 0.0, 5.0) as usize;
114        let col = clamp(col_raw, 0.0, 5.0) as usize;
115
116        code.push(DIGIPIN_GRID[row][col]);
117
118        // refine bounds
119        let new_max_y = min_y + y_div * (6.0 - row as f64);
120        let new_min_y = min_y + y_div * (5.0 - row as f64);
121
122        min_x = min_x + x_div * (col as f64);
123        let new_max_x = min_x + x_div;
124
125        min_y = new_min_y; max_y = new_max_y; max_x = new_max_x;
126    }
127
128    Ok(group_code(&code))
129}

131/// Decode a DIGIPIN back to the center lat/lon of its cell.
132pub fn get_lat_lng_from_digipin(digipin: &str) -> Result<LatLng, Error> {
133    let pin: String = digipin.chars().filter(|&ch| ch != '-').collect();
134    if pin.is_empty() { return Err(Error::InvalidPin); }
135
136    let MercBounds { mut min_x, mut max_x, mut min_y, mut max_y } = merc_bounds();
137
138    for ch in pin.chars() {
139        let (ri, ci) = lookup_grid(ch).ok_or(Error::InvalidChar(ch))?;
140
141        let x_div = (max_x - min_x) / 6.0;
142        let y_div = (max_y - min_y) / 6.0;
143
144        // rows counted from top
145        let y1 = max_y - y_div * ((ri + 1) as f64);
146        let y2 = max_y - y_div * (ri as f64);
147        let x1 = min_x + x_div * (ci as f64);
148        let x2 = x1 + x_div;
149
150        min_y = y1; max_y = y2;
151        min_x = x1; max_x = x2;
152    }
153
154    let cx = (min_x + max_x) / 2.0;
155    let cy = (min_y + max_y) / 2.0;
156
157    Ok(LatLng { latitude: y_to_lat(cy), longitude: x_to_lon(cx) })
158}

160/// Approximate cell size (meters) for a given code length.
161/// world width / 6^levels
162pub fn approx_cell_size_meters(levels: usize) -> f64 {
163    let world = 2.0 * PI * R;
164    world / 6f64.powi(levels as i32)
165}

167/// Group into "AAAA-BBBB-CC" when len==10; else groups of 4.
168fn group_code(raw: &str) -> String {
169    if raw.len() == 10 {
170        format!("{}-{}-{}", &raw[0..4], &raw[4..8], &raw[8..10])
171    } else {
172        let mut out = String::with_capacity(raw.len() + raw.len()/4);
173        for (i, ch) in raw.chars().enumerate() {
174            if i > 0 && i % 4 == 0 { out.push('-'); }
175            out.push(ch);
176        }
177        out
178    }
179}

181#[inline]
182fn lookup_grid(ch: char) -> Option<(usize, usize)> {
183    for (r, row) in DIGIPIN_GRID.iter().enumerate() {
184        for (c, &sym) in row.iter().enumerate() {
185            if sym == ch { return Some((r, c)); }
186        }
187    }
188    None
189}

195    #[test]
196    fn lon_normalization() {
197        assert_eq!(super::normalize_lon(180.0), -180.0);
198        assert!((super::normalize_lon(181.0) - -179.0).abs() < 1e-12);
199        assert!((super::normalize_lon(-181.0) - 179.0).abs() < 1e-12);
200    }

202    #[test]
203    fn approx_size_levels() {
204        let l5 = approx_cell_size_meters(5);
205        let l6 = approx_cell_size_meters(6);
206        assert!(l6 < l5);
207    }
208
209    #[test]
210    fn roundtrip_delhi_like() {
211        let lat = 28.6139_f64;
212        let lon = 77.2090_f64;
213
214        let code = get_digi_pin(lat, lon, 10).unwrap();
215        let ll = get_lat_lng_from_digipin(&code).unwrap();
216
217        let dist_lat = (ll.latitude - lat).abs();
218        let dist_lon = (ll.longitude - lon).abs();
219        assert!(dist_lat < 0.05);
220        assert!(dist_lon < 0.05);
221    }
